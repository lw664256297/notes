// MQTT 消息回调函数
void callback(char* topic, byte* payload, unsigned int length){

  // 字符转字符串---------------------用于开发环境调试代码
  // MQTT 消息定义  第一个数组为硬件类型， 第二个数组为开关
  Serial.print("Message arrived [");
  Serial.print(topic);
  Serial.print("] ");
  String data = "";
  for (int i = 0; i < length; i++) {
    Serial.print((char)payload[i]);
    data += String((char)payload[i]);
  }

  Serial.println();
  Serial.println("收到的信息:--------");
  Serial.print(data);

  // 执行控制函数
  car_control((char)payload[0]);

}

// MQTT 重新连接
void reconnect(){
  while(!client.connected()){
      Serial.print("Attempting MQTT connection...");

      // 创建一个随机客户端ID
     String clientId = "ESP8266Client-";
     clientId += String(random(0xffff), HEX);

     // 创建连接
     if (client.connect(clientId.c_str(), mqttUserName, mqttPassword, willTopic, willQoS, willRetain, willMessage, cleanSession)) {
      Serial.println("connected");
      // 发布
      client.publish("outTopic", "hello world");
      // 监听
      client.subscribe("inTopic");
     }else{
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" try again in 5 seconds");
      // Wait 5 seconds before retrying
      delay(5000);

     }
  }
}

void setup() {
  // SH74HC595N 扩展模块
  pinMode(big, OUTPUT);
  pinMode(datain, OUTPUT);
  pinMode(push, OUTPUT);

  pinMode(kkjled, OUTPUT);

  Serial.begin(115200);

  // 使用esp连接设置的wifi
  setuo_wifi();

  // 设置MQTT 服务器详细信息
  client.setServer(mqttServer, 1883);

  // 设置MQTT 消息回调函数
  client.setCallback(callback);

}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop();

}
